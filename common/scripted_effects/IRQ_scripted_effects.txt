IRQ_agrarian_society_modifier_level_down = {
	IF = {
		limit = {
			OR = {
				has_idea = IRQ_agrarian_society
				has_idea = IRQ_agrarian_society_2
				has_idea = IRQ_agrarian_society_3
				has_idea = IRQ_agrarian_society_4
			}
		}
		custom_effect_tooltip = IRQ_agrarian_society_modifier_level_down_tt
	}
	
	IF = {
		limit = {
			has_idea = IRQ_agrarian_society
		}
		swap_ideas = {
			remove_idea = IRQ_agrarian_society
			add_idea = IRQ_agrarian_society_2
		}	
	}
	ELSE_IF = {
		limit = {
			has_idea = IRQ_agrarian_society_2
		}
		swap_ideas = {
			remove_idea = IRQ_agrarian_society_2
			add_idea = IRQ_agrarian_society_3
		}
	}
	ELSE_IF = {
		limit = {
			has_idea = IRQ_agrarian_society_3
		}
		swap_ideas = {
			remove_idea = IRQ_agrarian_society_3
			add_idea = IRQ_agrarian_society_4
		}
	}
	ELSE_IF = {
		limit = {
			has_idea = IRQ_agrarian_society_4
		}
		remove_ideas = IRQ_agrarian_society_4
	}
}

IRQ_integrate_kurdish_state_effect = {
	custom_effect_tooltip = IRQ_integrate_kurdish_state_effect_tt
	hidden_effect = {
		event_target:IRQ_kurdish_state_event_target = {
			if = {
				limit = {
					event_target:IRQ_kurdish_state_event_target = { 
						NOT = { is_core_of = IRQ }
						compliance > 70
					}
				}
				event_target:IRQ_kurdish_state_event_target = { add_core_of = IRQ }
				remove_dynamic_modifier = { modifier = separatist_fatigue }
				remove_dynamic_modifier = { modifier = kurdish_agitation }
				remove_dynamic_modifier = { modifier = kurdish_separatism }
				remove_dynamic_modifier = { modifier = kurdish_rebellion }
			}
			if = {
				limit = {
					AND = {
						1047 = { is_core_of = IRQ }
						676 = { is_core_of = IRQ }
						has_idea = IRQ_sectarian_woes
					}
				}
				swap_ideas = {
					remove_idea = IRQ_sectarian_woes
					add_idea = IRQ_sectarian_woes_1
				}
			}
			else_if = {
				limit = {
					AND = {
						1047 = { is_core_of = IRQ }
						676 = { is_core_of = IRQ }
						has_idea = IRQ_sectarian_woes_2
					}
				}
				remove_ideas = IRQ_sectarian_woes_2
			}
		}
	}
}

# Kurdish State Integration Forced
IRQ_integrate_kurdish_state_forced_effect = {
	custom_effect_tooltip = IRQ_integrate_kurdish_state_effect_tt
	hidden_effect = {
		event_target:IRQ_kurdish_state_event_target = { 
			add_core_of = IRQ

			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = separatist_fatigue
					}
				}
				remove_dynamic_modifier = { modifier = separatist_fatigue }
			}

			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = kurdish_agitation
					}
				}
				remove_dynamic_modifier = { modifier = kurdish_agitation }
			}

			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = kurdish_separatism
					}
				}
				remove_dynamic_modifier = { modifier = kurdish_separatism }
			}

			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = kurdish_rebellion
					}
				}
				remove_dynamic_modifier = { modifier = kurdish_rebellion }
			}
		}
	}
}

			}
		}
	}
}
