add_namespace = ksr_news

# Formation of Grand Duchy of Lithuania
news_event = {
	id = ksr_news.1
	title = ksr_news.1.t
	desc = ksr_news.1.desc
	picture = GFX_news_event_generic_parliament

	is_triggered_only = yes
	major = yes
	
	option = {
		name = ksr_news.1.a
		trigger = {
			OR = {
				original_tag = SMG
				original_tag = LIT
			}
			NOT = { has_war_with = FROM }
		}
	}
	option = {
		name = ksr_news.1.b
		trigger = {
			NOT = { original_tag = SMG }
			NOT = { original_tag = LIT }
			NOT = { has_war_with = FROM }
		}
	}
	option = {
		name = ksr_news.1.c
		trigger = {
			has_war_with = FROM
		}
	}
}
